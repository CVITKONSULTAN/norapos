# Crontab entry untuk otomatis fix log permissions
# Tambahkan ke crontab dengan: sudo crontab -e
#
# Fix log permissions setiap 5 menit
*/5 * * * * /var/www/kartika/fix-log-permissions.sh > /dev/null 2>&1
